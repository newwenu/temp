根据屏幕尺寸和当前图标数量自动计算位置，按网格排列并整体居中：

```
计算步骤：
1. item_width = DESKTOP_ICON_SIZE + DESKTOP_ICON_SPACING
2. item_height = DESKTOP_ICON_SIZE + DESKTOP_ICON_TEXT_SIZE + DESKTOP_ICON_SPACING
3. icons_per_row = desktop_width / item_width
4. col = index % icons_per_row
5. row = index / icons_per_row
6. grid_width = icons_per_row * DESKTOP_ICON_SIZE + (icons_per_row - 1) * DESKTOP_ICON_SPACING
7. start_x = (desktop_width - grid_width) / 2  // 水平居中
8. start_y = DESKTOP_STATUSBAR_HEIGHT + DESKTOP_ICON_SPACING
9. x = start_x + col * item_width
10. y = start_y + row * item_height
```

#### 颜色插值算法

```c
static uint16_t desktop_interpolate_color(uint16_t color1, uint16_t color2, uint8_t ratio)
{
    uint8_t r1, g1, b1;
    uint8_t r2, g2, b2;
    uint8_t r, g, b;
    
    /* 提取RGB分量 */
    r1 = (color1 >> 11) & 0x1F;
    g1 = (color1 >> 5) & 0x3F;
    b1 = color1 & 0x1F;
    
    r2 = (color2 >> 11) & 0x1F;
    g2 = (color2 >> 5) & 0x3F;
    b2 = color2 & 0x1F;
    
    /* 线性插值 */
    r = r1 + ((r2 - r1) * ratio >> 8);
    g = g1 + ((g2 - g1) * ratio >> 8);
    b = b1 + ((b2 - b1) * ratio >> 8);
    
    return (r << 11) | (g << 5) | b;
}
```

### 1.8 时钟显示

#### 时间获取

```c
static void desktop_get_time_str(char *buf, uint8_t len)
{
    uint32_t tick = HAL_GetTick();                  /* 获取系统运行毫秒数 */
    uint32_t total_minutes = tick / 60000;          /* 换算为分钟 */
    uint32_t hours = (total_minutes / 60) % 24;     /* 小时 (0-23) */
    uint32_t minutes = total_minutes % 60;          /* 分钟 (0-59) */
    snprintf(buf, len, "%02lu:%02lu", hours, minutes);
}
```

**注意**：显示的是系统运行时间，从 00:00 开始递增，非实时时钟。

#### 时间更新优化

```c
static void desktop_update_time_display(void)
{
    char time_str[6];
    desktop_get_time_str(time_str, sizeof(time_str));

    /* 时间未变化时不重绘，避免闪烁和性能浪费 */
    if (strcmp(time_str, g_last_time_str) == 0)
    {
        return;
    }
    strcpy(g_last_time_str, time_str);

    /* 更新顶部状态栏时间 */
    lcd_show_string_bg(10, 8, 60, 16, 16, time_str, WHITE, BLACK);

    /* 更新底部任务栏时间 */
    uint16_t taskbar_y = desktop_height - DESKTOP_TASKBAR_HEIGHT;
    lcd_show_string_bg(desktop_width - 60, taskbar_y + 12, 50, 16, 16, time_str, WHITE, TASKBAR_BG_COLOR);
}
```

**优化点**：仅在分钟变化时重绘，避免每秒刷新造成的闪烁和性能浪费。

### 1.9 触摸处理

```c
nav_event_t desktop_handle_touch(uint16_t x, uint16_t y)
{
    uint8_t icon_index;
    uint16_t nav_y = desktop_height - DESKTOP_TASKBAR_HEIGHT;
    uint16_t third = desktop_width / 3;
    nav_event_t event = NAV_EVENT_NONE;

    /* 检查是否点击底部导航栏 */
    if (y >= nav_y)
    {
        if (x < third)
            event = NAV_EVENT_BACK;
        else if (x < third * 2)
            event = NAV_EVENT_HOME;
        else
            event = NAV_EVENT_RECENT;

        if (g_nav_callback != NULL)
            g_nav_callback(event);

        return event;
    }

    /* 检查是否点击图标 */
    icon_index = desktop_get_icon_at(x, y);
    if (icon_index != 0xFF)
        desktop_select_icon(icon_index);
    else
        desktop_clear_selection();

    return NAV_EVENT_NONE;
}
```

## 3. 开机画面 (startup_screen)

### 3.1 概述

开机画面在系统启动时显示，**不依赖外部字库**，仅使用 `lcd.h` 基础绘图和 ASCII 字模接口。实现位于 [`User/startup_screen.c`](file:///g:/codeversion/h732worktask/基于FreeRTOS的作业实现/User/startup_screen.c)。

### 3.2 设计约束

开机画面在 `main()` 中于 `fonts_init()` 之前调用，因此：
- 不能使用 `text_show_string()` 等依赖外部字库的函数
- 只能使用 `lcd_show_char()` 等内嵌 ASCII 字模的函数
- 不能使用中文显示

### 3.3 实现

```c
void startup_screen_show(void)
{
    uint16_t cx = lcddev.width / 2;
    uint16_t cy = lcddev.height / 2;

    /* 1. 清屏 */
    lcd_clear(BG_COLOR);

    /* 2. 绘制中央面板背景 */
    uint16_t pw = 320;
    uint16_t ph = 180;
    uint16_t px = cx - pw / 2;
    uint16_t py = cy - ph / 2;
    lcd_fill(px, py, px + pw, py + ph, PANEL_COLOR);

    /* 3. 面板边框 */
    lcd_draw_rectangle(px, py, px + pw, py + ph, BORDER_COLOR);
    lcd_draw_rectangle(px + 1, py + 1, px + pw - 1, py + ph - 1, BORDER_COLOR);

    /* 4. 标题（ASCII内嵌字模，不依赖外部字库，透明背景） */
    show_string_no_bg(cx - 100, py + 30, 24, "FreeOS Desktop", TITLE_COLOR);

    /* 5. 副标题 */
    show_string_no_bg(cx - 60, py + 70, 16, "System Booting...", TEXT_COLOR);

    /* 6. 进度条底框 */
    uint16_t bw = 240;
    uint16_t bh = 12;
    uint16_t bx = cx - bw / 2;
    uint16_t by = py + 120;
    lcd_fill(bx, by, bx + bw, by + bh, BAR_BG_COLOR);
    lcd_draw_rectangle(bx - 1, by - 1, bx + bw + 1, by + bh + 1, BORDER_COLOR);

    /* 7. 进度条填充（静态示意，分几段快速填充，不阻塞） */
    uint16_t seg = bw / 6;
    for (uint8_t i = 0; i < 5; i++)
    {
        lcd_fill(bx + 2 + i * seg, by + 2, bx + 2 + (i + 1) * seg - 1, by + bh - 2, BAR_FILL_COLOR);
    }

    /* 8. 版本号 */
    show_string_no_bg(cx - 40, py + 150, 12, "Ver 1.0", TEXT_COLOR);
}
```

### 3.4 透明背景字符串显示

```c
static void show_string_no_bg(uint16_t x, uint16_t y, uint8_t size, const char *str, uint16_t color)
{
    while (*str >= ' ' && *str <= '~')
    {
        lcd_show_char(x, y, *str, size, 1, color);
        x += size / 2;
        str++;
    }
}
```

- `mode=1` 表示透明背景，不覆盖原有像素
- 仅支持 ASCII 字符（' ' ~ '~'）

## 5. 本章小结

本章介绍了 FreeOS Desktop 的桌面系统、标题栏组件、开机画面和字符串管理。关键要点：

1. **桌面系统**：安卓风格，包含渐变壁纸、状态栏、导航栏、图标网格
2. **自动布局**：图标按网格排列，自动计算位置并整体居中
3. **时钟优化**：仅在分钟变化时重绘，避免闪烁和性能浪费
4. **导航栏**：返回（三角形）、主页（圆形）、最近任务（方形）
5. **标题栏**：统一的应用标题栏，包含返回按钮和标题文字
6. **开机画面**：不依赖外部字库，使用 ASCII 内嵌字模
7. **字符串集中管理**：所有字符串定义在 `app_strings.h`，使用 GB2312 编码
