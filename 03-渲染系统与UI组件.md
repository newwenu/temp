# FreeOS Desktop 技术手册 —— 第三部分：渲染系统与 UI 组件

## 1. 渲染命令机制

### 1.1 设计目标

渲染命令机制（Render Command）的设计目标是**批量收集绘制命令，一次性提交到 LCD**，减少频繁的 LCD 刷新操作，提升渲染效率并避免画面闪烁。

实现位于 [`Middlewares/RENDER/render_cmd.c`](file:///g:/codeversion/h732worktask/基于FreeRTOS的作业实现/Middlewares/RENDER/render_cmd.c) 和 [`Middlewares/RENDER/render_cmd.h`](file:///g:/codeversion/h732worktask/基于FreeRTOS的作业实现/Middlewares/RENDER/render_cmd.h)。

### 1.2 核心设计

```c
#define RCMD_MAX_CMDS   256     /* 最大命令数 */
#define RCMD_STR_POOL   1024    /* 字符串池大小 */

typedef enum
{
    RCMD_FILL,          /* 填充矩形 */
    RCMD_DRAW_RECT,     /* 绘制矩形边框 */
    RCMD_FILL_CIRCLE,   /* 填充圆 */
    RCMD_DRAW_CIRCLE,   /* 绘制圆边框 */
    RCMD_DRAW_LINE,     /* 绘制直线 */
    RCMD_DRAW_HLINE,    /* 绘制水平线 */
    RCMD_TEXT,          /* 显示文本（英文/数字） */
    RCMD_TEXT_ZH,       /* 显示中文文本 */
    RCMD_TEXT_BG,       /* 显示带背景色的文本 */
    RCMD_CLEAR,         /* 清屏 */
    RCMD_COLOR_FILL,    /* 颜色填充（用于图片数据） */
    RCMD_DRAW_POINT,    /* 绘制点 */
} rcmd_type_t;

typedef struct
{
    uint16_t    p1;         /* 参数1（通常为x1或cx） */
    uint16_t    p2;         /* 参数2（通常为y1或cy） */
    uint16_t    p3;         /* 参数3（通常为x2或r） */
    uint16_t    p4;         /* 参数4（通常为y2） */
    uint16_t    color;      /* 前景色 */
    uint16_t    bgcolor;    /* 背景色 */
    uint8_t     cmd;        /* 命令类型 */
    uint8_t     font_size;  /* 字体大小 */
    const char *str;        /* 字符串指针 */
} rcmd_t;
```

### 1.3 命令缓冲区

```c
static rcmd_t g_cmds[RCMD_MAX_CMDS];        /* 命令缓冲区 */
static uint16_t g_cmd_count = 0;             /* 当前命令数 */
static uint8_t g_overflow = 0;               /* 溢出标志 */

static char g_str_pool[RCMD_STR_POOL];       /* 字符串池 */
static uint16_t g_str_pool_pos = 0;          /* 字符串池位置 */
```

### 1.4 使用模式

渲染命令采用**开始 → 添加命令 → 提交**的三段式使用模式：

```c
rcmd_begin();
/* 多条绘制命令 */
rcmd_fill(0, DESKTOP_STATUSBAR_HEIGHT, lcddev.width - 1,
          lcddev.height - DESKTOP_TASKBAR_HEIGHT - 1, 0x1082);
rcmd_text(50, 100, 200, 24, 16, "Hello, FreeOS!", WHITE);
rcmd_flush();   // 一次性提交到 LCD
```

### 1.5 API 详解

#### rcmd_begin() — 开始批量绘制

```c
void rcmd_begin(void)
{
    g_cmd_count = 0;
    g_overflow = 0;
    g_str_pool_pos = 0;
}
```

重置命令计数器和字符串池，准备接收新的一批绘制命令。

#### rcmd_flush() — 提交并执行

```c
void rcmd_flush(void)
{
    uint16_t i;
    for (i = 0; i < g_cmd_count; i++)
    {
        const rcmd_t *c = &g_cmds[i];
        switch ((rcmd_type_t)c->cmd)
        {
        case RCMD_FILL:
            lcd_fill(c->p1, c->p2, c->p3, c->p4, c->color);
            break;
        case RCMD_DRAW_RECT:
            lcd_draw_rectangle(c->p1, c->p2, c->p3, c->p4, c->color);
            break;
        case RCMD_FILL_CIRCLE:
            lcd_fill_circle(c->p1, c->p2, c->p3, c->color);
            break;
        case RCMD_DRAW_CIRCLE:
            lcd_draw_circle(c->p1, c->p2, (uint8_t)c->p3, c->color);
            break;
        case RCMD_DRAW_LINE:
            lcd_draw_line(c->p1, c->p2, c->p3, c->p4, c->color);
            break;
        case RCMD_DRAW_HLINE:
            lcd_draw_hline(c->p1, c->p2, c->p3, c->color);
            break;
        case RCMD_DRAW_POINT:
            lcd_draw_point(c->p1, c->p2, c->color);
            break;
        case RCMD_TEXT:
            lcd_show_string(c->p1, c->p2, c->p3, c->p4, c->font_size, (char *)c->str, c->color);
            break;
        case RCMD_TEXT_BG:
            lcd_show_string_bg(c->p1, c->p2, c->p3, c->p4, c->font_size, (char *)c->str, c->color, c->bgcolor);
            break;
        case RCMD_TEXT_ZH:
            text_show_string(c->p1, c->p2, c->p3, c->p4, (char *)c->str, c->font_size, 1, c->color);
            break;
        case RCMD_CLEAR:
            lcd_clear(c->color);
            break;
        case RCMD_COLOR_FILL:
            lcd_color_fill(c->p1, c->p2, c->p3, c->p4, (uint16_t *)c->str);
            break;
        default:
            break;
        }
    }
    g_cmd_count = 0;
    g_overflow = 0;
    g_str_pool_pos = 0;
}
```

遍历所有缓冲的命令并调用对应的 LCD 底层函数执行绘制，最后清空缓冲区。

### 1.6 命令添加函数

所有命令添加函数遵循相同的模式：分配命令槽 → 填充参数 → 返回。

```c
void rcmd_fill(uint16_t x1, uint16_t y1, uint16_t x2, uint16_t y2, uint16_t color)
{
    rcmd_t *c = rcmd_add(RCMD_FILL);
    if (c) { c->p1 = x1; c->p2 = y1; c->p3 = x2; c->p4 = y2; c->color = color; }
}

void rcmd_text(uint16_t x, uint16_t y, uint16_t w, uint16_t h, uint8_t size, const char *str, uint16_t color)
{
    rcmd_t *c = rcmd_add(RCMD_TEXT);
    if (c) { c->p1 = x; c->p2 = y; c->p3 = w; c->p4 = h; c->font_size = size; c->str = rcmd_strcpy(str); c->color = color; }
}
```

### 1.7 字符串池管理

```c
static const char *rcmd_strcpy(const char *str)
{
    if (str == NULL) return "";

    uint16_t len = strlen(str) + 1;
    if (g_str_pool_pos + len > RCMD_STR_POOL) return "";

    const char *ret = &g_str_pool[g_str_pool_pos];
    memcpy(&g_str_pool[g_str_pool_pos], str, len);
    g_str_pool_pos += len;
    return ret;
}
```

- 字符串被复制到内部池中，避免外部字符串生命周期问题
- 池满时返回空字符串，不阻塞

### 1.8 溢出处理

```c
static rcmd_t *rcmd_add(rcmd_type_t type)
{
    if (g_cmd_count >= RCMD_MAX_CMDS)
    {
        g_overflow = 1;
        return NULL;
    }
    rcmd_t *c = &g_cmds[g_cmd_count++];
    memset(c, 0, sizeof(rcmd_t));
    c->cmd = (uint8_t)type;
    return c;
}
```

- 命令缓冲区满时设置溢出标志，返回 NULL
- 调用者通过检查返回值判断是否成功

### 1.9 使用场景对比

| 场景 | 使用方式 | 说明 |
|------|----------|------|
| 应用 UI 绘制 | `rcmd_begin()` → 多条命令 → `rcmd_flush()` | 批量绘制，减少刷新次数 |
| ISR 中（如 camera DMA） | 直接调用 `ltdc_color_fill()` | ISR 中不使用 rcmd，直接操作 LCD |
| 简单单条绘制 | 可直接调用 LCD 底层函数 | 无需批量的场景 |

### 2.2 数据项组件 (ui_data_item)

数据项组件实现位于 [`User/APP/Components/ui_data_item.c`](file:///g:/codeversion/h732worktask/基于FreeRTOS的作业实现/User/APP/Components/ui_data_item.c) 和 [`User/APP/Components/ui_data_item.h`](file:///g:/codeversion/h732worktask/基于FreeRTOS的作业实现/User/APP/Components/ui_data_item.h)。

#### 设计目标

- 一行数据展示：左侧静态标签 + 右侧动态数值
- 支持**局部刷新**：仅数值变化时重绘数值区域

#### 数据结构

```c
typedef struct
{
    uint16_t x;              /* 整行左上角 X */
    uint16_t y;              /* 整行左上角 Y */
    uint16_t label_w;        /* 标签区域宽度 */
    uint16_t val_w;          /* 数值区域宽度 */
    uint16_t h;              /* 行高 */
    uint16_t bg_color;       /* 背景色（用于擦除） */
    uint16_t label_color;    /* 标签文本颜色 */
    uint16_t val_color;      /* 数值文本颜色 */
    uint8_t  font_size;      /* 字体大小 */
    const char *label;       /* 静态标签文本 */
    char     val_str[16];    /* 当前数值字符串 */
    uint8_t  dirty;          /* 1 = 数值已变化，需要重绘 */
    uint8_t  drawn;          /* 1 = 已首次绘制（标签已画过） */
} ui_data_item_t;
```

#### 初始化

```c
void ui_data_item_init(ui_data_item_t *item,
                       uint16_t x, uint16_t y,
                       uint16_t label_w, uint16_t val_w, uint16_t h,
                       uint16_t bg_color,
                       uint16_t label_color, uint16_t val_color,
                       uint8_t font_size,
                       const char *label)
```

#### 设置数值

```c
void ui_data_item_setf(ui_data_item_t *item, const char *fmt, ...)
{
    char new_str[16];
    va_list ap;

    va_start(ap, fmt);
    vsnprintf(new_str, sizeof(new_str), fmt, ap);
    va_end(ap);

    /* 仅当内容变化时才标记 dirty，避免无意义重绘 */
    if (strncmp(item->val_str, new_str, sizeof(item->val_str)) != 0)
    {
        strncpy(item->val_str, new_str, sizeof(item->val_str));
        item->dirty = 1;
    }
}
```

**优化点**：通过比较新旧值，仅在内容变化时标记 `dirty`，避免无意义重绘。

#### 绘制

```c
void ui_data_item_draw(ui_data_item_t *item)
{
    uint16_t val_x = item->x + item->label_w + 5;
    uint16_t val_y = item->y + (item->h - item->font_size) / 2;

    if (!item->drawn)
    {
        /* 首次：绘制标签 + 数值 */
        rcmd_fill(item->x, item->y,
                  item->x + item->label_w + item->val_w - 1,
                  item->y + item->h - 1,
                  item->bg_color);

        rcmd_text(item->x + 2, val_y,
                  item->label_w, item->font_size + 2,
                  item->font_size, item->label, item->label_color);

        item->drawn = 1;
    }

    if (item->dirty)
    {
        /* 仅刷新数值区域 */
        rcmd_fill(val_x, item->y,
                  val_x + item->val_w - 1,
                  item->y + item->h - 1,
                  item->bg_color);

        rcmd_text(val_x, val_y,
                  item->val_w, item->font_size + 2,
                  item->font_size, item->val_str, item->val_color);

        item->dirty = 0;
    }
}
```

**绘制策略**：

- **首次绘制**：绘制背景、标签和数值
- **后续绘制**：仅当 `dirty=1` 时重绘数值区域
- **局部刷新**：只重绘数值区域，不触碰标签

```c
uint8_t ui_data_row_group_add(ui_data_row_group_t *grp, const char *label)
{
    uint16_t item_y;

    if (grp == NULL || grp->row_cnt >= UI_ROW_GROUP_MAX_ROWS)
    {
        return 0;
    }

    /* 行内上边距 4px，依次向下排列 */
    item_y = grp->y + 4 + grp->row_cnt * grp->row_h;

    ui_data_item_init(&grp->items[grp->row_cnt],
                      grp->x + 10, item_y,
                      grp->label_w, grp->val_w, grp->row_h,
                      grp->bg_color,
                      grp->label_color, grp->val_color,
                      grp->font_size, label);

    grp->row_cnt++;
    return 1;
}
```

**自动布局**：行内上边距 4px，每行依次向下排列，左边距 10px。

#### 绘制

```c
void ui_data_row_group_draw(ui_data_row_group_t *grp)
{
    uint16_t border_h;
    uint8_t  i;

    border_h = 4 + grp->row_cnt * grp->row_h + 4;

    if (!grp->drawn)
    {
        /* 首次：背景 + 边框 */
        rcmd_fill(grp->x, grp->y,
                  grp->x + grp->w - 1,
                  grp->y + border_h - 1,
                  grp->bg_color);

        rcmd_draw_rect(grp->x, grp->y,
                       grp->x + grp->w - 1,
                       grp->y + border_h - 1,
                       grp->border_color);

        /* 绘制所有行 */
        for (i = 0; i < grp->row_cnt; i++)
        {
            ui_data_item_draw(&grp->items[i]);
        }

        grp->drawn = 1;
    }
    else
    {
        /* 后续：仅刷新数值发生变化的行 */
        for (i = 0; i < grp->row_cnt; i++)
        {
            ui_data_item_draw(&grp->items[i]);
        }
    }
}
```

**绘制策略**：

- **首次绘制**：绘制背景、边框和所有行
- **后续绘制**：仅刷新 `dirty=1` 的行（由 `ui_data_item_draw` 内部判断）

#### 获取底部坐标

```c
uint16_t ui_data_row_group_get_bottom(const ui_data_row_group_t *grp)
{
    uint16_t border_h = 4 + grp->row_cnt * grp->row_h + 4;
    return grp->y + border_h - 1;
}
```

## 3. 内存池管理

### 3.1 内存池架构

系统实现了 **6 个独立内存池**，分别对应 STM32H743 的不同内存区域。实现位于 [`Middlewares/MALLOC/malloc.c`](file:///g:/codeversion/h732worktask/基于FreeRTOS的作业实现/Middlewares/MALLOC/malloc.c) 和 [`Middlewares/MALLOC/malloc.h`](file:///g:/codeversion/h732worktask/基于FreeRTOS的作业实现/Middlewares/MALLOC/malloc.h)。

### 3.2 内存池配置

| 内存池 | 内存区域 | 物理地址 | 块大小 | 最大容量 | 分配表大小 |
|--------|---------|----------|--------|----------|-----------|
| MEM1 (SRAMIN) | AXI SRAM | 内部 | 64 字节 | 448 KB | 7168 |
| MEM2 (SRAMEX) | SDRAM | 0xC01F4000 | 64 字节 | 28,912 KB | 462592 |
| MEM3 (SRAM12) | SRAM1+SRAM2 | 0x30000000 | 64 字节 | 240 KB | 3840 |
| MEM4 (SRAM4) | SRAM4 | 0x38000000 | 64 字节 | 60 KB | 960 |
| MEM5 (SRAMDTCM) | DTCM | 0x20000000 | 64 字节 | 120 KB | 1920 |
| MEM6 (SRAMITCM) | ITCM | 0x00000000 | 64 字节 | 60 KB | 960 |

### 3.3 内存分配算法

采用**块分配表**方式管理内存：

```c
#define MT_TYPE     uint32_t

struct _m_mallco_dev
{
    void (*init)(uint8_t);                              /* 初始化 */
    uint16_t (*perused)(uint8_t);                       /* 内存使用率 */
    uint8_t *membase[SRAMBANK];                         /* 内存池基地址 */
    uint32_t *memmap[SRAMBANK];                         /* 内存分配状态表 */
    uint8_t  memrdy[SRAMBANK];                          /* 内存池是否就绪 */
};
```

- 每个内存池分为固定大小的块（64 字节）
- 分配表记录每个块的状态：0 表示空闲，非 0 表示已分配（值为连续分配的块数）
- 分配时从后向前搜索连续的空闲块

### 3.4 内存分配流程

```c
uint32_t my_mem_malloc(uint8_t memx, uint32_t size)
{
    signed long offset = 0;
    uint32_t nmemb;         /* 需要分配的块数 */
    uint32_t cmemb = 0;     /* 连续空闲块计数 */
    uint32_t i;

    if (!mallco_dev.memrdy[memx])
    {
        mallco_dev.init(memx);      /* 未初始化则先初始化 */
    }

    if (size == 0)
    {
        return 0xFFFFFFFF;          /* 不需要分配 */
    }

    nmemb = size / memblksize[memx];
    if (size % memblksize[memx])
    {
        nmemb++;
    }

    for (offset = memtblsize[memx] - 1; offset >= 0; offset--)
    {
        if (!mallco_dev.memmap[memx][offset])
        {
            cmemb++;                /* 连续空闲块数增加 */
        }
        else
        {
            cmemb = 0;              /* 连续中断，清零 */
        }

        if (cmemb == nmemb)         /* 找到足够的连续块 */
        {
            for (i = 0; i < nmemb; i++)
            {
                mallco_dev.memmap[memx][offset + i] = nmemb;
            }
            return (offset * memblksize[memx]);
        }
    }

    return 0xFFFFFFFF;              /* 分配失败 */
}
```

### 3.5 内存释放

```c
uint8_t my_mem_free(uint8_t memx, uint32_t offset)
{
    uint32_t i;

    if (!mallco_dev.memrdy[memx])
    {
        mallco_dev.init(memx);
        return 1;
    }

    if (offset < memsize[memx])
    {
        uint32_t index = offset / memblksize[memx];
        uint32_t nmemb = mallco_dev.memmap[memx][index];

        for (i = 0; i < nmemb; i++)
        {
            mallco_dev.memmap[memx][index + i] = 0;
        }
        return 0;
    }

    return 2;
}
```

### 3.6 内存使用率

```c
uint16_t my_mem_perused(uint8_t memx)
{
    uint32_t used = 0;
    uint32_t i;

    for (i = 0; i < memtblsize[memx]; i++)
    {
        if (mallco_dev.memmap[memx][i])
        {
            used++;
        }
    }

    return (used * 1000) / (memtblsize[memx]);
}
```

返回值为千分比（0~1000），例如 500 表示 50.0%。

### 3.7 使用建议

| 内存池 | 适用场景 |
|--------|----------|
| SRAMIN (AXI) | 一般动态分配 |
| SRAMEX (SDRAM) | 大缓冲区、图片数据、帧缓冲 |
| SRAM12 | 中等大小数据 |
| SRAM4 | 小数据、临时缓冲区 |
| SRAMDTCM | 需要 CPU 快速访问的数据 |
| SRAMITCM | 需要 CPU 快速访问的代码/数据 |

## 4. 本章小结

本章介绍了 FreeOS Desktop 的渲染命令机制、UI 组件系统和内存池管理。关键要点：

1. **渲染命令机制**：批量收集绘制命令，一次性提交，减少 LCD 刷新次数
2. **字符串池**：内部管理字符串生命周期，避免外部字符串失效问题
3. **数据项组件**：支持局部刷新，仅数值变化时重绘，提升效率
4. **数据行分组**：自动布局，首次绘制边框+所有行，后续仅刷新变化行
5. **按钮组件**：支持按下状态、颜色自定义、触摸检测
6. **6 个独立内存池**：充分利用 STM32H743 的多区域内存架构
7. **块分配算法**：64 字节固定块大小，从后向前搜索连续空闲块
